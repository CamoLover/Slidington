Get Started

Button Models Showcase

Default Model

slidington.create({
    title: 'Visit Portfolio',
    description: 'Click to explore amazing projects',
    image: '#555555',
    link: 'https://camolover.dev',
    buttonColor: '#3a8d8d',
    arrowColor: '#ffde00'
});

Triangle Model

slidington.create({
    title: 'Take Action Portfolio',
    description: 'Click to trigger custom function',
    model: 'triangle',
    image: 'path/to/image.jpg', 
    action: 'showAlert',
    buttonColor: '#23542a',
    arrowColor: '#ffde00'
});

Solid Model

slidington.create({
    title: 'Visit NPM',
    description: 'Click to explore projects NPM',
    model: 'solid',
    image: '#442759',
    link: 'https://www.npmjs.com/package/slidington',
    buttonColor: '#c477fc',
    arrowColor: '#141652'
});
