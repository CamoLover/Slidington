Button Models Showcase

Default Model

slidington.create({
    title: 'Visit Portfolio',
    description: 'Click to explore amazing projects',
    image: '#555555',
    link: 'https://camolover.dev',
    buttonColor: '#3a8d8d',
    arrowColor: '#ffde00'
});

    action: 'showAlert',

});
